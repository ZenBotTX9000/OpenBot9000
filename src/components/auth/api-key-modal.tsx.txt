"use client";

import { useApiKey } from "@/hooks/use-api-key";
import { motion } from "framer-motion";
import { useState } from "react";
import { HelpCircle, KeyRound, LoaderCircle } from "lucide-react";

export function ApiKeyModal() {
  const [key, setKey] = useState("");
  const { saveApiKey } = useApiKey();
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setIsLoading(true);
    setError(null);

    // Mock validation - in a real app, this would be a lightweight API call
    // to OpenRouter's /auth/key endpoint.
    await new Promise(resolve => setTimeout(resolve, 1000));
    if (key.startsWith("sk-or-")) {
      saveApiKey(key);
    } else {
      setError("Invalid API key format. It should start with 'sk-or-'.");
      setIsLoading(false);
    }
  };

  return (
    <div className="fixed inset-0 flex items-center justify-center p-4 backdrop-blur-sm" style={{ background: 'rgb(var(--color-grey-200) / 0.5)' }}>
      <motion.div
        className="w-full max-w-md p-8 shape-chamfer backdrop-blur-xl"
        style={{ background: 'var(--gradient-surface-primary)' }}
        initial={{ opacity: 0, scale: 0.9, y: 20 }}
        animate={{ opacity: 1, scale: 1, y: 0 }}
        transition={{ type: 'spring', damping: 20, stiffness: 150 }}
      >
        <h2 className="text-2xl font-bold text-center text-gradient" style={{ backgroundImage: 'var(--gradient-text-primary)' }}>
          Connect to OpenRouter
        </h2>
        <p className="mt-2 text-sm text-center text-gradient" style={{ backgroundImage: 'var(--gradient-text-secondary)' }}>
          Your key is stored securely in your browser and is never sent to our servers.
        </p>
        
        <form onSubmit={handleSubmit} className="mt-6 space-y-4">
          <div className="relative">
            <KeyRound className="absolute left-3 top-1/2 -translate-y-1/2 h-5 w-5 text-gray-400" />
            <input
              type="password"
              value={key}
              onChange={(e) => setKey(e.target.value)}
              placeholder="Paste your OpenRouter API key"
              className="w-full pl-10 pr-4 py-3 bg-transparent border-b-2 text-gradient focus:outline-none"
              style={{
                borderColor: 'rgb(var(--color-grey-800) / 0.2)',
                backgroundImage: 'var(--gradient-text-primary)',
              }}
            />
          </div>
          
          {error && <p className="text-sm text-red-500">{error}</p>}

          <motion.button
            type="submit"
            disabled={isLoading}
            className="w-full py-3 shape-chamfer font-semibold text-white flex items-center justify-center"
            style={{ background: 'var(--gradient-action)' }}
            whileHover={{ scale: 1.02, filter: 'brightness(1.1)' }}
            whileTap={{ scale: 0.98 }}
          >
            {isLoading ? <LoaderCircle className="animate-spin" /> : "Connect"}
          </motion.button>
        </form>

        <div className="mt-6 text-xs text-center">
            <details>
                <summary className="cursor-pointer inline-flex items-center text-gradient" style={{ backgroundImage: 'var(--gradient-text-secondary)'}}>
                    <HelpCircle className="h-4 w-4 mr-1"/> How to get your API key?
                </summary>
                <div className="mt-2 text-left space-y-1 p-2 rounded-md" style={{ background: 'rgb(var(--color-grey-200) / 0.5)' }}>
                    <p>1. Sign in to <a href="https://openrouter.ai" target="_blank" rel="noopener noreferrer" className="font-bold">OpenRouter.ai</a></p>
                    <p>2. Navigate to your 'Keys' page from the account menu.</p>
                    <p>3. Create a new key and paste it here.</p>
                </div>
            </details>
        </div>
      </motion.div>
    </div>
  );
}
